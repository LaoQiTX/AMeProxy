```rust
// 从 GitHub 下载核心二进制文件
pub async fn download_core(
    client: &reqwest::Client,
    core_type: &str,
    version: &str,
    artifact: &str,
) -> Result<Vec<u8>> {
    let url = format!(
        "https://github.com/{}/releases/download/{}/{}",
        get_repo_for_core(core_type),
        version,
        artifact
    );
    
    let response = client.get(&url).send().await?;
    let bytes = response.bytes().await?;
    Ok(bytes.to_vec())
}

// 保存到应用数据目录
pub fn save_core(bytes: Vec<u8>, core_type: &str) -> Result<PathBuf> {
    let data_dir = dirs::app_data_dir()?;
    let binary_path = data_dir.join(get_binary_name(core_type));
    std::fs::write(&binary_path, bytes)?;
    
    #[cfg(target_os = "windows")]
    { /* Windows 无需特殊权限 */ }
    
    #[cfg(target_os = "macos")]
    {
        // macOS 需要提权才能使用 TUN 模式
        grant_tun_permission(&binary_path)?;
    }
    
    #[cfg(target_os = "linux")]
    {
        // Linux 需要 setcap
        grant_capabilities(&binary_path)?;
    }
    
    Ok(binary_path)
}
```

#### 2.1.2 启动 Clash 核心

```rust
pub struct CoreInstance {
    core_type: CoreType,
    binary_path: PathBuf,
    config_path: PathBuf,
    child: Option<Child>,
}

impl CoreInstance {
    pub async fn start(&mut self) -> Result<()> {
        let mut command = Command::new(&self.binary_path);
        command
            .arg("-f")
            .arg(&self.config_path)
            .arg("-d")
            .arg(self.get_data_dir())
            .stdout(Stdio::piped())
            .stderr(Stdio::piped());
        
        let child = command.spawn()?;
        self.child = Some(child);
        
        // 等待端口就绪
        self.wait_for_ready().await?;
        Ok(())
    }
    
    pub async fn stop(&mut self) -> Result<()> {
        if let Some(mut child) = self.child.take() {
            child.kill().await?;
        }
        Ok(())
    }
}
```

```rust
pub struct ConfigBuilder {
    mapping: Mapping,
}

impl ConfigBuilder {
    pub fn new() -> Self {
        let mut mapping = Mapping::new();
        mapping.insert("mixed-port".into(), 7890.into());
        mapping.insert("log-level".into(), "info".into());
        mapping.insert("allow-lan".into(), false.into());
        mapping.insert("mode".into(), "rule".into());
        Self { mapping }
    }
    
    pub fn with_port(mut self, port: u16) -> Self {
        self.mapping.insert("mixed-port".into(), port.into());
        self
    }
    
    pub fn with_controller(mut self, host: &str, port: u16) -> Self {
        let addr = format!("{}:{}", host, port);
        self.mapping.insert("external-controller".into(), addr.into());
        self
    }
    
    pub fn with_proxies(mut self, proxies: Vec<ProxyConfig>) -> Self {
        let proxies: Vec<Value> = proxies.into_iter().map(|p| p.into()).collect();
        self.mapping.insert("proxies".into(), Value::Sequence(proxies));
        self
    }
    
    pub fn with_proxy_groups(mut self, groups: Vec<ProxyGroup>) -> Self {
        let groups: Vec<Value> = groups.into_iter().map(|g| g.into()).collect();
        self.mapping.insert("proxy-groups".into(), Value::Sequence(groups));
        self
    }
    
    pub fn with_rules(mut self, rules: Vec<Rule>) -> Self {
        let rules: Vec<Value> = rules.into_iter().map(|r| r.into()).collect();
        self.mapping.insert("rules".into(), Value::Sequence(rules));
        self
    }
    
    pub fn build(self) -> Mapping {
        self.mapping
    }
    
    pub async fn save(&self, path: &Path) -> Result<()> {
        let yaml = serde_yaml::to_string(&self.mapping)?;
        tokio::fs::write(path, yaml).await?;
        Ok(())
    }
}
```

### 2.3 REST API 客户端

#### 2.3.1 API 请求封装

```rust
pub struct ClashApiClient {
    server: String,
    secret: Option<String>,
    client: reqwest::Client,
}

impl ClashApiClient {
    pub fn new(server: &str, secret: Option<&str>) -> Self {
        Self {
            server: server.to_string(),
            secret: secret.map(|s| s.to_string()),
            client: reqwest::Client::new(),
        }
    }
    
    fn headers(&self) -> HeaderMap {
        let mut headers = HeaderMap::new();
        headers.insert("Content-Type", "application/json".parse().unwrap());
        if let Some(ref secret) = self.secret {
            let auth = format!("Bearer {}", secret).parse().unwrap();
            headers.insert("Authorization", auth);
        }
        headers
    }
    
    pub async fn get_proxies(&self) -> Result<ProxiesResponse> {
        let url = format!("{}/proxies", self.server);
        let resp = self.client
            .get(&url)
            .headers(self.headers())
            .send()
            .await?
            .json()
            .await?;
        Ok(resp)
    }
    
    pub async fn update_proxy(&self, group: &str, name: &str) -> Result<()> {
        let url = format!("{}/proxies/{}", self.server, group);
        let body = serde_json::json!({ "name": name });
        
        self.client
            .put(&url)
            .headers(self.headers())
            .json(&body)
            .send()
            .await?;
        Ok(())
    }
    
    pub async fn get_config(&self) -> Result<ConfigResponse> {
        let url = format!("{}/configs", self.server);
        let resp = self.client
            .get(&url)
            .headers(self.headers())
            .send()
            .await?
            .json()
            .await?;
        Ok(resp)
    }
    
    pub async fn reload_config(&self, path: &str) -> Result<()> {
        let url = format!("{}/configs", self.server);
        let body = serde_json::json!({ "path": path });
        
        self.client
            .put(&url)
            .headers(self.headers())
            .json(&body)
            .send()
            .await?;
        Ok(())
    }
    
    pub async fn get_proxy_delay(&self, name: &str, timeout: u64) -> Result<u64> {
        let url = format!(
            "{}/proxies/{}/delay?timeout={}&url=http://www.gstatic.com/generate_204",
            self.server, name, timeout
        );
        let resp: DelayResponse = self.client
            .get(&url)
            .headers(self.headers())
            .send()
            .await?
            .json()
            .await?;
        Ok(resp.delay)
    }
}
```

```rust
// 从订阅链接解析节点
pub fn parse_subscription(url: &str) -> Result<Vec<ProxyConfig>> {
    let content = reqwest::get(url).await?.text().await?;
    let decoded = decode_base64(&content)?;
    let proxies: Vec<ProxyConfig> = serde_yaml::from_str(&decoded)?;
    Ok(proxies)
}

// 从单节点链接解析
pub fn parse_node_link(link: &str) -> Result<ProxyConfig> {
    if link.starts_with("ss://") {
        parse_ss_link(link)
    } else if link.starts_with("vmess://") {
        parse_vmess_link(link)
    } else if link.starts_with("trojan://") {
        parse_trojan_link(link)
    } else if link.starts_with("vless://") {
        parse_vless_link(link)
    } else {
        anyhow::bail!("Unknown node link format")
    }
}

fn parse_ss_link(link: &str) -> Result<ProxyConfig> {
    let link = link.trim_start_matches("ss://");
    let (auth, server_info) = link.split_once('@').ok_or_else(|| anyhow::anyhow!("Invalid ss link"))?;
    let (server, port_part) = server_info.split_once(':').ok_or_else(|| anyhow::anyhow!("Invalid ss link"))?;
    let port: u16 = port_part.split_once('#').map(|(p, _)| p).unwrap_or(port_part).parse()?;
    
    let (cipher, password) = decode_base64(auth)?.split_once(':').ok_or_else(|| anyhow::anyhow!("Invalid ss auth"))?;
    
    Ok(ProxyConfig::Shadowsocks(ShadowsocksConfig {
        name: urlencoding_decode(server_info.split('#').last().unwrap_or(""))?,
        server: server.to_string(),
        port,
        cipher: cipher.to_string(),
        password: password.to_string(),
        udp: true,
    }))
}

fn parse_vmess_link(link: &str) -> Result<ProxyConfig> {
    let link = link.trim_start_matches("vmess://");
    let decoded = decode_base64(link)?;
    let json: serde_json::Value = serde_json::from_str(&decoded)?;
    
    Ok(ProxyConfig::Vmess(VmessConfig {
        name: json["ps"].as_str().unwrap_or("").to_string(),
        server: json["add"].as_str().ok_or_else(|| anyhow::anyhow!("Missing server"))?.to_string(),
        port: json["port"].as_str().unwrap_or("").parse()?,
        uuid: json["id"].as_str().ok_or_else(|| anyhow::anyhow!("Missing uuid"))?.to_string(),
        alter_id: json["aid"].as_str().unwrap_or("0").parse()?,
        cipher: json["scy"].as_str().unwrap_or("auto").to_string(),
        network: json["net"].as_str().unwrap_or("tcp").to_string(),
        tls: json["tls"].as_str() == "tls",
    }))
}
```

### 2.6 系统代理设置

#### 2.6.1 Windows

```rust
#[cfg(target_os = "windows")]
pub fn set_system_proxy(host: &str, port: u16) -> Result<()> {
    use std::ptr::null_mut;
    
    #[repr(C)]
    struct INTERNET_PROXY_INFO {
        dwAccessType: u32,
        lpszProxy: *const u16,
        lpszProxyBypass: *const u16,
    }
    
    let proxy = format!("http={}:{}", host, port);
    let proxy_bypass = "localhost;127.*;10.*;172.16.*;192.168.*";
    
    unsafe {
        let proxy_info = INTERNET_PROXY_INFO {
            dwAccessType: 3, // INTERNET_OPEN_TYPE_PROXY
            lpszProxy: to_wide_ptr(&proxy),
            lpszProxyBypass: to_wide_ptr(proxy_bypass),
        };
        
        winapi::um::wininet::InternetSetOptionW(
            null_mut(),
            38, // INTERNET_OPTION_PROXY
            &proxy_info as *const _ as *const _,
            std::mem::size_of::<INTERNET_PROXY_INFO>() as u32,
        );
    }
    
    Ok(())
}

#[cfg(target_os = "windows")]
pub fn reset_system_proxy() -> Result<()> {
    unsafe {
        winapi::um::wininet::InternetSetOptionW(
            null_mut(),
            39, // INTERNET_OPTION_PROXY_SETTINGS
            null_mut(),
            0,
        );
    }
    Ok(())
}
```

### 2.8 WebSocket 监控

```rust
pub struct ClashWebSocketClient {
    url: String,
    secret: Option<String>,
}

impl ClashWebSocketClient {
    pub fn new(server: &str, secret: Option<&str>) -> Self {
        let auth = secret.map(|s| format!("?token={}", s)).unwrap_or_default();
        let url = format!("ws://{}/connections{}", server.trim_start_matches("http://"), auth);
        Self { url, secret }
    }
    
    pub async fn connect(&self) -> Result<WebSocketStream<TokioTcpStream>> {
        let (stream, _) = tokio_tungstenite::connect_async(&self.url).await?;
        Ok(stream)
    }
    
    pub async fn subscribe<F>(&self, mut callback: F) -> Result<()>
    where
        F: FnMut(ConnectionInfo) + Send + 'static,
    {
        let mut stream = self.connect().await?;
        
        loop {
            if let Some(msg) = stream.next().await {
                match msg? {
                    Message::Text(text) => {
                        let info: ConnectionInfo = serde_json::from_str(&text)?;
                        callback(info);
                    }
                    Message::Close(_) => break,
                    _ => {}
                }
            }
        }
        
        Ok(())
    }
}

#[derive(Debug, Clone, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct ConnectionInfo {
    pub download_total: u64,
    pub upload_total: u64,
    pub connections: Vec<Connection>,
}

#[derive(Debug, Clone, Deserialize)]
pub struct Connection {
    pub id: String,
    pub meta: ConnectionMeta,
    pub upload: u64,
    pub download: u64,
}

#[derive(Debug, Clone, Deserialize)]
```

### 4.2 最小示例

```rust
use anyhow::Result;
use std::path::PathBuf;

#[tokio::main]
async fn main() -> Result<()> {
    // 1. 下载核心
    let core_path = download_mihomo().await?;
    
    // 2. 生成配置
    let config_path = generate_config()?;
    
    // 3. 启动核心
    let mut instance = CoreInstance::new(&core_path, &config_path);
    instance.start().await?;
    
    // 4. 设置系统代理
    set_system_proxy("127.0.0.1", 7890)?;
    
    println!("代理已启动: 127.0.0.1:7890");
    
    // 5. 保持运行
    tokio::signal::ctrl_c().await?;
    
    // 6. 清理
    reset_system_proxy()?;
    instance.stop().await?;
    
    Ok(())
}
```
